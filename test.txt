fof(php3, conjecture, (
    % Every pigeon (1, 2, 3, 4) must be in at least one hole (1, 2, 3)
    ((p11 | p12 | p13) & (p21 | p22 | p23) & (p31 | p32 | p33) & (p41 | p42 | p43))
    =>
    % There must be a collision (Two pigeons in the same hole)
    (
        (p11 & p21) | (p11 & p31) | (p11 & p41) | (p21 & p31) | (p21 & p41) | (p31 & p41) | % Hole 1
        (p12 & p22) | (p12 & p32) | (p12 & p42) | (p22 & p32) | (p22 & p42) | (p32 & p42) | % Hole 2
        (p13 & p23) | (p13 & p33) | (p13 & p43) | (p23 & p33) | (p23 & p43) | (p33 & p43)   % Hole 3
    )
)).
