fof(easy4, conjecture, (p => ~~p)).

fof(easy6, conjecture, (![X] : p(X) => p(a))).

fof(pelletier_21, conjecture, (((p => q) & (m => n) & (p | m)) => (q | n))).

fof(de_morgan_quantifier, conjecture, (~(![X] : p(X)) => (?[X] : ~p(X)))).

fof(quantifier_dist_and, conjecture, (![X] : (p(X) & q(X)) => (![X] : p(X) & ![X] : q(X)))).

% Hard

fof(graph_colouring, conjecture,
  (((r1 | g1 | b1) & (r2 | g2 | b2) & (r3 | g3 | b3)) &
   (~(r1 & r2)) & (~(g1 & g2)) & (~(b1 & b2)) &
   (~(r1 & r3)) & (~(g1 & g3)) & (~(b1 & b3)) &
   (~(r2 & r3)) & (~(g2 & g3)) & (~(b2 & b3)))
  =>
  ((r1 | g1 | b1) & (r2 | g2 | b2) & (r3 | g3 | b3))).
